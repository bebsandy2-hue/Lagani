<script>

let balance = Number(localStorage.getItem("bal") || 0);
let vip = Number(localStorage.getItem("vip") || 0);
let history = JSON.parse(localStorage.getItem("his") || "[]");

const incomeTable = [0,5,15,40,80,150];

function update(){
document.getElementById("balance").innerText = "$" + balance;
document.getElementById("vipLevel").innerText = "VIP " + vip;
document.getElementById("income").innerText = "Daily Income: $" + incomeTable[vip];
document.getElementById("history").innerHTML =
history.slice(-6).reverse().map(h=>"• "+h).join("<br>");
}

function save(){
localStorage.setItem("bal",balance);
localStorage.setItem("vip",vip);
localStorage.setItem("his",JSON.stringify(history));
update();
}

function addMoney(){
balance += 100;
history.push("+Rs100 Demo Added");
save();
}

function resetAll(){
balance=0;vip=0;history=[];
save();
}

function deposit(){
let a=Number(document.getElementById("dAmt").value);
if(a>0){
balance+=a;
history.push("Deposited Rs"+a);
save();
alert("Success");
}
}

function withdraw(){
let a=Number(document.getElementById("wAmt").value);
if(a>0 && a<=balance){
balance-=a;
history.push("Withdraw $"+a);
save();
alert("Request Sent");
}else alert("Insufficient balance");
}

function upgradeVIP(){
if(vip<5){
vip++;
history.push("Upgraded to VIP "+vip);
save();
}else{
alert("Max VIP reached");
}
}

function showDeposit(){
document.getElementById("depositBox").style.display="block";
document.getElementById("withdrawBox").style.display="none";
}

function showWithdraw(){
document.getElementById("withdrawBox").style.display="block";
document.getElementById("depositBox").style.display="none";
}

update();

</script>
